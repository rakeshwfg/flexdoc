Simple to Use

const { FlexDoc } = require('flexdoc');
const flexdoc = new FlexDoc();

// Convert HTML to PDF
const pdfResult = await flexdoc.toPDF(htmlContent, {
  outputPath: './output.pdf',
  format: 'A4'
});

// Convert HTML to PPTX
const pptxResult = await flexdoc.toPPTX(htmlContent, {
  outputPath: './presentation.pptx',
  splitBy: 'h2'
});
// Enable Adobe-quality professional mode
const result = await flexdoc.toPPTX(htmlContent, {
  professional: true,           // Enable AI optimization
  theme: 'corporate',          // Professional template
  optimizeLayout: true,        // Smart layout engine
  convertTablesToCharts: true, // Auto visualization
  enhanceImages: true,         // Image processing
  outputPath: './professional.pptx'
});

// AI automatically:
// ✓ Optimizes layout and visual balance
// ✓ Converts data tables to charts
// ✓ Applies professional design principles
// ✓ Creates stunning presentations
// Batch conversion with progress tracking
const documents = [...]; // Your HTML documents

const results = await flexdoc.batchConvert(documents.map(doc => ({
  html: doc.content,
  options: {
    format: doc.type === 'report' ? 'pdf' : 'pptx',
    professional: true,
    onProgress: (progress) => {
      console.log(`${progress.percentage}% - ${progress.step}`);
    }
  }
})));

// With retry logic for reliability
const result = await flexdoc.convertWithRetry(html, options, 3);
